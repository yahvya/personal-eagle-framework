632 public static function buildPrimaryKeysCondOn(MysqlModel $model,MysqlQueryBuilder $queryBuilder,bool $addWhere = true):MysqlQueryBuilder{
633 if($addWhere)
634 $queryBuilder->where();
635 else
636 $queryBuilder->cond(MysqlCondSeparator::AND());
637
638 // ajout du groupe de conditions
639 $columnsConfig = $model->getColumnsConfig();
640 $primaryKeysCond = [];
641
642 foreach($columnsConfig as $attributeName => $columnConfig){
643 if($columnConfig->isPrimaryKey()){
644 $primaryKeysCond[] = new MysqlCondition(
645 condGetter: $attributeName,
646 comparator: MysqlComparator::EQUAL(),
647 conditionValue: $columnConfig->convertFromValue(data: $model->$attributeName)
648 );
649 $primaryKeysCond[] = MysqlCondSeparator::AND();
650 }
651 }
652
653 if(empty($primaryKeysCond) )
654 throw new MysqlException(message: "Aucune clé primaire trouvée",isDisplayable: false);
655
656 // remplacement de la dernière condition AND
657 $primaryKeysCond[count(value: $primaryKeysCond) - 1] = MysqlCondSeparator::GROUP_END();
658
659 return $queryBuilder->cond(
660 MysqlCondSeparator::GROUP_START(),
661 ...$primaryKeysCond
662 );
